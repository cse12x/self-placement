public class Example2 {
    public static void main(String[] args) {
        figure(8);
    }

    public static void figure(int max) {
        for (int line = 1; line <= max; line++) {
            for (int dots = 1; dots <= -1 * line + max; dots++) {
                System.out.print(".");
            }
            System.out.println(line);
        }
    }
}
